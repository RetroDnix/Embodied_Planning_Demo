```python
def purchase_item(item: str, location: str, initial_position: str):
    """确保机器人醒来，然后去指定地点购买物品并返回初始位置。

    Examples:
        purchase_item("cola", "shop", "initial_position")
    """
    # Stage 1: 确保机器人醒着
    if not check("robot_awake"):
        wake_up()  # 唤醒机器人
    
    # Stage 2: 找到指定的地点并前往该地点
    find_and_go_to_location(location)  # 寻找并到达指定地点
    
    # Stage 3: 购买指定物品
    purchase_item(item, "cashier")  # 在收银台购买物品
    
    # Stage 4: 返回初始位置
    go_to(initial_position)  # 回到机器人初始位置
```
